basic 10 container network cluster jaringan
docker network create --driver bridge mongonetwork

docker container create --name mongodbnetw --network mongonetwork --env MONGO_INITDB_ROOT_USERNAME=root --env MONGO_INITDB_ROOT_PASSWORD=root mongo:latest

docker run -d --name mongoakses `
  --network mongonetwork `
  -p 8081:8081 `
  -e ME_CONFIG_MONGODB_URL="mongodb://root:root@mongodbnetw:27017/" `
  -e ME_CONFIG_BASICAUTH_USERNAME=root `
  -e ME_CONFIG_BASICAUTH_PASSWORD=root `
  mongo-express:latest


docker container start mongodbnetw
docker container start mongoakses

untuk memutus atau menghubungkan koneksi network 
docker network disconnect mongonetwork mongodbnetw
docker network connect mongonetwork mongodbnetw
